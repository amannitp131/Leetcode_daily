//https://leetcode.com/problems/maximum-points-you-can-obtain-from-cards/

class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int l = k - 1, r = cardPoints.size() - 1, m = 0, s = 0;
        for (int i = 0; i < k; i++) {

            m = m + cardPoints[i];
        }
        s = m;
        while (l >= 0) {
            s = s - cardPoints[l] + cardPoints[r];
            m = max(s, m);
            l--;
            r--;
        }
        return m;
    }
};
